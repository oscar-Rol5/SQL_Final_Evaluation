CREATE DATABASE EBDS;
USE EBDS;

CREATE TABLE IF NOT EXISTS STUDENT(
	NUMSTUD INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    NAME VARCHAR(50) NOT NULL,
    FIRSTNAME VARCHAR(50) NOT NULL,
    DATEOFBIRTH DATE NOT NULL,
    STREET VARCHAR(50) NOT NULL,
    POSTALCOD VARCHAR(50) NOT NULL,
    CITY VARCHAR(50) NOT NULL
	
);

CREATE TABLE IF NOT EXISTS MATTER(
	CODEMAT VARCHAR(50) PRIMARY KEY NOT NULL,
	WORDING VARCHAR(50) NOT NULL,
    COEF FLOAT NOT NULL

);

CREATE TABLE IF NOT EXISTS TEST(
	NUMTEST INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    TESTDATE DATE NOT NULL,
    PLACE VARCHAR(50) NOT NULL,
    CODEMAT VARCHAR(50),
    
    CONSTRAINT CODEMAT FOREIGN KEY (CODEMAT) REFERENCES MATTER(CODEMAT)

);

CREATE TABLE IF NOT EXISTS NOTATION(
	NUMSTUD INT,
    NUMTEST INT,
    SCORE FLOAT NOT NULL,
    
    CONSTRAINT NUMSTUD FOREIGN KEY (NUMSTUD) REFERENCES STUDENT(NUMSTUD),
	CONSTRAINT NUMTEST FOREIGN KEY (NUMTEST) REFERENCES TEST(NUMTEST)
    
);
